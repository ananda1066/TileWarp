package com.weebly.httptilewarp.tilewarp; 
 
import android.content.Intent; 
import android.content.SharedPreferences; 
import android.os.CountDownTimer; 
import android.support.v7.app.ActionBarActivity; 
import android.os.Bundle; 
import android.view.Menu; 
import android.view.MenuItem; 
import android.view.View; 
import android.widget.Button; 
import android.widget.ImageButton; 
import android.widget.TextView; 
 
 
public class SplashScreen9Intermediate extends ActionBarActivity { 
 
 
    TextView direction; 
    CountDownTimer cdt; 
    ImageButton nextlevel; 
    ImageButton surrender; 
 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_splash_screen9_intermediate); 
 
        nextlevel = (ImageButton) findViewById(R.id.nextlevel); 
        nextlevel.setOnClickListener(new View.OnClickListener() { 
            @Override 
            public void onClick(View v) { 
                startActivity(new Intent(SplashScreen9Intermediate.this, TileWarpIntermediateLevel10.class)); 
                finish(); 
            } 
        }); 
 
        surrender = (ImageButton) findViewById(R.id.surrender); 
        surrender.setOnClickListener(new View.OnClickListener() { 
            @Override 
            public void onClick(View v) { 
                startActivity(new Intent(SplashScreen9Intermediate.this, MainActivity.class)); 
                finish(); 
            } 
        }); 
 
        direction = (TextView) findViewById(R.id.direction); 
 
 
    } 
 
    public boolean isrunning; 
 
    @Override 
    protected void onStart() { 
        super.onStart(); 
 
        // Store our shared preference 
        SharedPreferences sp = getSharedPreferences("OURINFO", MODE_PRIVATE); 
        SharedPreferences.Editor ed = sp.edit(); 
        ed.putBoolean("active", true); 
        ed.apply(); 
        isrunning = true; 
    } 
 
    @Override 
    protected void onStop() { 
        super.onStop(); 
 
        // Store our shared preference 
        SharedPreferences sp = getSharedPreferences("OURINFO", MODE_PRIVATE); 
        SharedPreferences.Editor ed = sp.edit(); 
        ed.putBoolean("active", false); 
        ed.apply(); 
        isrunning  = false; 
 
    } 
}
